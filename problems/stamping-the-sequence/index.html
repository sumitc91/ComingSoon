936. Stamping The Sequence

Jump to Solution: Python Java C++ JavaScript C#

Problem Description

You want to form a target string of lowercase letters.

At the beginning, your sequence is target.length '?' marks. You also have a stamp of lowercase letters.

On each turn, you may place the stamp over the sequence, and replace every letter in the sequence with the corresponding letter from the stamp. You can make up to 10 * target.length turns.

For example, if the initial sequence is "?????", and your stamp is "abc", then you may make "abc??", "?abc?", "??abc" in the first turn. (Note that the stamp must be fully contained within the boundaries of the sequence in order to stamp.)

If the sequence is possible to stamp, then return an array of the index of the left-most letter being stamped at each turn. If the sequence is not possible to stamp, return an empty array.

For example, if the sequence is "ababc", and the stamp is "abc", then we could return the answer [0, 2], corresponding to the moves "?????" -> "abc??" -> "ababc".

Also, if the sequence is possible to stamp, it is guaranteed it is possible to stamp within 10 * target.length moves. Any answers specifying more than 10 * target.length moves will not be accepted.

Example 1:

Input: stamp = "abc", target = "ababc"
Output: [0,2]
Explanation: Initially s = "?????".
- Place stamp at index 0 to get "abc??".
- Place stamp at index 2 to get "ababc".
[1,0,2] would also be accepted as an answer, as well as some other answers.

Example 2:

Input: stamp = "abca", target = "aabcaca"
Output: [3,0,1]
Explanation: Initially s = "???????".
- Place stamp at index 3 to get "???abca".
- Place stamp at index 0 to get "abcabca".
- Place stamp at index 1 to get "aabcaca".

Constraints:

  • 1 <= stamp.length <= target.length <= 1000
  • stamp and target only contain lowercase letters.

Python Solution

class Solution:
    def movesToStamp(self, stamp: str, target: str) -> List[int]:
        def canStamp(start: int) -> bool:
            for i in range(len(stamp)):
                if target[start + i] != '?' and target[start + i] != stamp[i]:
                    return False
            return True
        
        def stampAt(start: int) -> int:
            count = 0
            for i in range(len(stamp)):
                if target[start + i] != '?':
                    count += 1
                    target[start + i] = '?'
            return count
        
        target = list(target)
        n = len(target)
        m = len(stamp)
        result = []
        total = 0
        
        while total < n:
            found = False
            for i in range(n - m + 1):
                if canStamp(i):
                    found = True
                    result.append(i)
                    total += stampAt(i)
                    break
            if not found:
                return []
                
        return result[::-1]

Time Complexity: O(n * (n - m))

Where n is the length of target and m is the length of stamp.

Space Complexity: O(n)

We need to store the target string as a list.

Java Solution

class Solution {
    private boolean canStamp(String target, String stamp, int start) {
        for (int i = 0; i < stamp.length(); i++) {
            if (target.charAt(start + i) != '?' && target.charAt(start + i) != stamp.charAt(i)) {
                return false;
            }
        }
        return true;
    }
    
    private int stampAt(StringBuilder target, String stamp, int start) {
        int count = 0;
        for (int i = 0; i < stamp.length(); i++) {
            if (target.charAt(start + i) != '?') {
                count++;
                target.setCharAt(start + i, '?');
            }
        }
        return count;
    }
    
    public int[] movesToStamp(String stamp, String target) {
        StringBuilder sb = new StringBuilder(target);
        int n = target.length();
        int m = stamp.length();
        List result = new ArrayList<>();
        int total = 0;
        
        while (total < n) {
            boolean found = false;
            for (int i = 0; i <= n - m; i++) {
                if (canStamp(sb.toString(), stamp, i)) {
                    found = true;
                    result.add(i);
                    total += stampAt(sb, stamp, i);
                    break;
                }
            }
            if (!found) {
                return new int[0];
            }
        }
        
        int[] ans = new int[result.size()];
        for (int i = 0; i < ans.length; i++) {
            ans[i] = result.get(result.size() - 1 - i);
        }
        return ans;
    }
}

public class Solution {
    private bool CanStamp(string target, string stamp, int start) {
        for (int i = 0; i < stamp.Length; i++) {
            if (target[start + i] != '?' && target[start + i] != stamp[i]) {
                return false;
            }
        }
        return true;
    }
    
    private int StampAt(StringBuilder target, string stamp, int start) {
        int count = 0;
        for (int i = 0; i < stamp.Length; i++) {
            if (target[start + i] != '?') {
                count++;
                target[start + i] = '?';
            }
        }
        return count;
    }
    
    public int[] MovesToStamp(string stamp, string target) {
        StringBuilder sb = new StringBuilder(target);
        int n = target.Length;
        int m = stamp.Length;
        List result = new List();
        int total = 0;
        
        while (total < n) {
            bool found = false;
            for (int i = 0; i <= n - m; i++) {
                if (CanStamp(sb.ToString(), stamp, i)) {
                    found = true;
                    result.Add(i);
                    total += StampAt(sb, stamp, i);
                    break;
                }
            }
            if (!found) {
                return new int[0];
            }
        }
        
        result.Reverse();
        return result.ToArray();
    }
}
