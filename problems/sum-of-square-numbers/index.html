633. Sum of Square Numbers

Jump to Solution: Python Java C++ JavaScript C#

Problem Description

Given a non-negative integer c, decide whether there're two integers a and b such that a² + b² = c.

Examples:

Example 1:

Input: c = 5
Output: true
Explanation: 1² + 2² = 5

Example 2:

Input: c = 3
Output: false

Example 3:

Input: c = 4
Output: true
Explanation: 0² + 2² = 4

Constraints:

  • 0 ≤ c ≤ 231 - 1

Python Solution

class Solution:
    def judgeSquareSum(self, c: int) -> bool:
        left, right = 0, int(sqrt(c))
        
        while left <= right:
            curr = left * left + right * right
            if curr == c:
                return True
            elif curr < c:
                left += 1
            else:
                right -= 1
                
        return False

Alternative Solution (Using Set):

class Solution:
    def judgeSquareSum(self, c: int) -> bool:
        squares = set()
        i = 0
        
        while i * i <= c:
            squares.add(i * i)
            if c - i * i in squares:
                return True
            i += 1
            
        return False

Implementation Notes:

  • First solution uses two pointers: O(sqrt(c)) time, O(1) space
  • Second solution uses set: O(sqrt(c)) time, O(sqrt(c)) space
  • Both handle edge cases including 0
  • Two pointers solution is more space efficient

Java Solution

class Solution {
    public boolean judgeSquareSum(int c) {
        long left = 0;
        long right = (long) Math.sqrt(c);
        
        while (left <= right) {
            long curr = left * left + right * right;
            if (curr == c) {
                return true;
            } else if (curr < c) {
                left++;
            } else {
                right--;
            }
        }
        
        return false;
    }
}

Alternative Solution (Using Binary Search):

class Solution {
public:
    bool judgeSquareSum(int c) {
        for (long a = 0; a * a <= c; a++) {
            long b = c - a * a;
            if (binary_search(0, b, b)) {
                return true;
            }
        }
        return false;
    }
    
private:
    bool binary_search(long s, long e, long n) {
        if (s > e) return false;
        
        long mid = s + (e - s) / 2;
        if (mid * mid == n) return true;
        if (mid * mid > n) return binary_search(s, mid - 1, n);
        return binary_search(mid + 1, e, n);
    }
};
