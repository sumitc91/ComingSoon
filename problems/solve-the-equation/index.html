640. Solve the Equation

Jump to Solution: Python Java C++ JavaScript C#

Problem Description

Solve a given equation and return the value of x in the form of a string "x=#value". The equation contains only '+', '-' operation, the variable x and its coefficient.

If there is no solution for the equation, return "No solution". If there are infinite solutions, return "Infinite solutions".

If there is exactly one solution for the equation, we ensure that the value of x is an integer.

Examples:

Example 1:

Input: equation = "x+5-3+x=6+x-2"
Output: "x=2"

Example 2:

Input: equation = "x=x"
Output: "Infinite solutions"

Example 3:

Input: equation = "2x=x"
Output: "x=0"

Constraints:

  • 3 ≤ equation.length ≤ 1000
  • equation has exactly one '='
  • equation consists of integers with an absolute value in the range [0, 100] without any leading zeros, and the variable 'x'

Python Solution

class Solution:
    def solveEquation(self, equation: str) -> str:
        def parse(s: str) -> tuple[int, int]:
            x_coef = const = 0
            num = 0
            sign = 1
            i = 0
            
            while i < len(s):
                if s[i] == '+':
                    sign = 1
                    i += 1
                elif s[i] == '-':
                    sign = -1
                    i += 1
                else:
                    # Parse number
                    j = i
                    while j < len(s) and (s[j].isdigit() or s[j] == 'x'):
                        j += 1
                    
                    term = s[i:j]
                    if term == 'x':
                        x_coef += sign
                    elif term.endswith('x'):
                        x_coef += sign * int(term[:-1])
                    else:
                        const += sign * int(term)
                    
                    i = j
                    
            return x_coef, const
        
        # Split equation into left and right sides
        left, right = equation.split('=')
        
        # Parse both sides
        left_x, left_const = parse(left)
        right_x, right_const = parse(right)
        
        # Move all terms to left side
        x_coef = left_x - right_x
        const = right_const - left_const
        
        if x_coef == 0 and const == 0:
            return "Infinite solutions"
        if x_coef == 0:
            return "No solution"
            
        return f"x={const // x_coef}"

Java Solution

class Solution {
    private static class Term {
        int xCoef;
        int constant;
        
        Term(int xCoef, int constant) {
            this.xCoef = xCoef;
            this.constant = constant;
        }
    }
    
    private Term parse(String s) {
        int xCoef = 0, constant = 0;
        int num = 0;
        int sign = 1;
        int i = 0;
        
        while (i < s.length()) {
            char c = s.charAt(i);
            if (c == '+') {
                sign = 1;
                i++;
            } else if (c == '-') {
                sign = -1;
                i++;
            } else {
                // Parse number
                StringBuilder term = new StringBuilder();
                while (i < s.length() && (Character.isDigit(s.charAt(i)) || s.charAt(i) == 'x')) {
                    term.append(s.charAt(i));
                    i++;
                }
                
                String termStr = term.toString();
                if (termStr.equals("x")) {
                    xCoef += sign;
                } else if (termStr.endsWith("x")) {
                    xCoef += sign * Integer.parseInt(termStr.substring(0, termStr.length() - 1));
                } else {
                    constant += sign * Integer.parseInt(termStr);
                }
            }
        }
        
        return new Term(xCoef, constant);
    }
    
    public String solveEquation(String equation) {
        String[] parts = equation.split("=");
        Term left = parse(parts[0]);
        Term right = parse(parts[1]);
        
        int xCoef = left.xCoef - right.xCoef;
        int constant = right.constant - left.constant;
        
        if (xCoef == 0 && constant == 0) {
            return "Infinite solutions";
        }
        if (xCoef == 0) {
            return "No solution";
        }
        
        return "x=" + (constant / xCoef);
    }
}

JavaScript Solution

/**
 * @param {string} equation
 * @return {string}
 */
var solveEquation = function(equation) {
    function parse(s) {
        let xCoef = 0, constant = 0;
        let sign = 1;
        let i = 0;
        
        while (i < s.length) {
            if (s[i] === '+') {
                sign = 1;
                i++;
            } else if (s[i] === '-') {
                sign = -1;
                i++;
            } else {
                // Parse number
                let j = i;
                while (j < s.length && (s[j].match(/[0-9x]/) !== null)) {
                    j++;
                }
                
                const term = s.slice(i, j);
                if (term === 'x') {
                    xCoef += sign;
                } else if (term.endsWith('x')) {
                    xCoef += sign * parseInt(term.slice(0, -1));
                } else {
                    constant += sign * parseInt(term);
                }
                
                i = j;
            }
        }
        
        return { xCoef, constant };
    }
    
    const [left, right] = equation.split('=');
    const leftTerm = parse(left);
    const rightTerm = parse(right);
    
    const xCoef = leftTerm.xCoef - rightTerm.xCoef;
    const constant = rightTerm.constant - leftTerm.constant;
    
    if (xCoef === 0 && constant === 0) {
        return "Infinite solutions";
    }
    if (xCoef === 0) {
        return "No solution";
    }
    
    return `x=${Math.floor(constant / xCoef)}`;
};

C# Solution

public class Solution {
    private class Term {
        public int XCoef { get; set; }
        public int Constant { get; set; }
        
        public Term(int xCoef = 0, int constant = 0) {
            XCoef = xCoef;
            Constant = constant;
        }
    }
    
    private Term Parse(string s) {
        var result = new Term();
        int sign = 1;
        int i = 0;
        
        while (i < s.Length) {
            if (s[i] == '+') {
                sign = 1;
                i++;
            } else if (s[i] == '-') {
                sign = -1;
                i++;
            } else {
                // Parse number
                int j = i;
                while (j < s.Length && (char.IsDigit(s[j]) || s[j] == 'x')) {
                    j++;
                }
                
                string term = s.Substring(i, j - i);
                if (term == "x") {
                    result.XCoef += sign;
                } else if (term.EndsWith("x")) {
                    result.XCoef += sign * int.Parse(term.Substring(0, term.Length - 1));
                } else {
                    result.Constant += sign * int.Parse(term);
                }
                
                i = j;
            }
        }
        
        return result;
    }
    
    public string SolveEquation(string equation) {
        var parts = equation.Split('=');
        var left = Parse(parts[0]);
        var right = Parse(parts[1]);
        
        int xCoef = left.XCoef - right.XCoef;
        int constant = right.Constant - left.Constant;
        
        if (xCoef == 0 && constant == 0) {
            return "Infinite solutions";
        }
        if (xCoef == 0) {
            return "No solution";
        }
        
        return $"x={constant / xCoef}";
    }
}

Alternative Solution (Using LINQ):

public class Solution {
    public string SolveEquation(string equation) {
        var sides = equation.Split('=')
            .Select(side => {
                // Add '+' before negative terms for consistent splitting
                side = side.Replace("-", "+-");
                if (side.StartsWith("+")) side = side.Substring(1);
                
                var terms = side.Split('+').Where(t => t.Length > 0);
                var xCoef = terms.Where(t => t.Contains('x'))
                    .Sum(t => t == "x" ? 1 : 
                           t == "-x" ? -1 : 
                           int.Parse(t.TrimEnd('x')));
                var constant = terms.Where(t => !t.Contains('x'))
                    .Sum(t => int.Parse(t));
                    
                return (xCoef, constant);
            })
            .ToArray();
            
        int xCoef = sides[0].xCoef - sides[1].xCoef;
        int constant = sides[1].constant - sides[0].constant;
        
        if (xCoef == 0 && constant == 0) return "Infinite solutions";
        if (xCoef == 0) return "No solution";
        
        return $"x={constant / xCoef}";
    }
}
