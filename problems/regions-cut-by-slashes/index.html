959. Regions Cut By Slashes

Jump to Solution: Python Java C++ JavaScript C#

Problem Description

In a N x N grid composed of 1 x 1 squares, each 1 x 1 square consists of a /, \, or blank space. These characters divide the square into contiguous regions.

(Note that backslash characters are escaped, so a \ is represented as "\\".)

Return the number of regions.

Solution

Python Solution

class Solution:
    def regionsBySlashes(self, grid: List[str]) -> int:
        n = len(grid)
        # Each cell is divided into 4 triangles
        parent = [i for i in range(4 * n * n)]
        rank = [0] * (4 * n * n)
        
        def find(x):
            if parent[x] != x:
                parent[x] = find(parent[x])
            return parent[x]
        
        def union(x, y):
            px, py = find(x), find(y)
            if px != py:
                if rank[px] < rank[py]:
                    px, py = py, px
                parent[py] = px
                if rank[px] == rank[py]:
                    rank[px] += 1
        
        for i in range(n):
            for j in range(n):
                base = 4 * (i * n + j)
                if grid[i][j] == '/':
                    union(base, base + 3)
                    union(base + 1, base + 2)
                elif grid[i][j] == '\\':
                    union(base, base + 1)
                    union(base + 2, base + 3)
                else:
                    union(base, base + 1)
                    union(base + 1, base + 2)
                    union(base + 2, base + 3)
                
                # Connect to right cell
                if j < n - 1:
                    union(base + 1, base + 7)
                
                # Connect to bottom cell
                if i < n - 1:
                    union(base + 2, base + 4 * n)
        
        return sum(1 for i in range(4 * n * n) if parent[i] == i)

Time Complexity: O(n²)

Where n is the size of the grid. We need to process each cell and perform union operations.

Space Complexity: O(n²)

We need to store the parent and rank arrays for each of the 4 triangles in each cell.

Java Solution

class Solution {
    public int regionsBySlashes(String[] grid) {
        int n = grid.length;
        // Each cell is divided into 4 triangles
        int[] parent = new int[4 * n * n];
        int[] rank = new int[4 * n * n];
        
        for (int i = 0; i < parent.length; i++) {
            parent[i] = i;
        }
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                int base = 4 * (i * n + j);
                char c = grid[i].charAt(j);
                
                if (c == '/') {
                    union(parent, rank, base, base + 3);
                    union(parent, rank, base + 1, base + 2);
                } else if (c == '\\') {
                    union(parent, rank, base, base + 1);
                    union(parent, rank, base + 2, base + 3);
                } else {
                    union(parent, rank, base, base + 1);
                    union(parent, rank, base + 1, base + 2);
                    union(parent, rank, base + 2, base + 3);
                }
                
                // Connect to right cell
                if (j < n - 1) {
                    union(parent, rank, base + 1, base + 7);
                }
                
                // Connect to bottom cell
                if (i < n - 1) {
                    union(parent, rank, base + 2, base + 4 * n);
                }
            }
        }
        
        int count = 0;
        for (int i = 0; i < parent.length; i++) {
            if (parent[i] == i) {
                count++;
            }
        }
        return count;
    }
    
    private int find(int[] parent, int x) {
        if (parent[x] != x) {
            parent[x] = find(parent, parent[x]);
        }
        return parent[x];
    }
    
    private void union(int[] parent, int[] rank, int x, int y) {
        int px = find(parent, x);
        int py = find(parent, y);
        
        if (px != py) {
            if (rank[px] < rank[py]) {
                int temp = px;
                px = py;
                py = temp;
            }
            parent[py] = px;
            if (rank[px] == rank[py]) {
                rank[px]++;
            }
        }
    }
}

        return parent[x];

    void union(vector& parent, vector& rank, int x, int y) {
        int px = find(parent, x);
        int py = find(parent, y);

            if (rank[px] < rank[py]) {
                swap(px, py);
            }
            parent[py] = px;

                rank[px]++;
            }

JavaScript Solution

/**
 * @param {string[]} grid
 * @return {number}
 */
var regionsBySlashes = function(grid) {
    const n = grid.length;
    // Each cell is divided into 4 triangles
    const parent = new Array(4 * n * n);
    const rank = new Array(4 * n * n).fill(0);
    
    for (let i = 0; i < parent.length; i++) {
        parent[i] = i;
    }
    
    function find(x) {
        if (parent[x] !== x) {
            parent[x] = find(parent[x]);
        }
        return parent[x];
    }
    
    function union(x, y) {
        const px = find(x);
        const py = find(y);
        
        if (px !== py) {
            if (rank[px] < rank[py]) {
                [px, py] = [py, px];
            }
            parent[py] = px;
            if (rank[px] === rank[py]) {
                rank[px]++;
            }
        }
    }
    
    for (let i = 0; i < n; i++) {
        for (let j = 0; j < n; j++) {
            const base = 4 * (i * n + j);
            const c = grid[i][j];
            
            if (c === '/') {
                union(base, base + 3);
                union(base + 1, base + 2);
            } else if (c === '\\') {
                union(base, base + 1);
                union(base + 2, base + 3);
            } else {
                union(base, base + 1);
                union(base + 1, base + 2);
                union(base + 2, base + 3);
            }
            
            // Connect to right cell
            if (j < n - 1) {
                union(base + 1, base + 7);
            }
            
            // Connect to bottom cell
            if (i < n - 1) {
                union(base + 2, base + 4 * n);
            }
        }
    }
    
    return parent.filter((p, i) => p === i).length;
};
