576. Out of Boundary Paths

Jump to Solution: Python Java C++ JavaScript C#

Problem Description

There is an m x n grid with a ball. The ball is initially at the position [startRow, startColumn]. You are allowed to move the ball to one of the four adjacent cells in the grid (possibly out of the grid crossing the grid boundary). You can apply at most maxMove moves to the ball.

Given the five integers m, n, maxMove, startRow, startColumn, return the number of paths to move the ball out of the grid boundary. Since the answer can be very large, return it modulo 10‚Åπ + 7.

Example 1:

Input: m = 2, n = 2, maxMove = 2, startRow = 0, startColumn = 0
Output: 6

Example 2:

Input: m = 1, n = 3, maxMove = 3, startRow = 0, startColumn = 1
Output: 12

Constraints:

  • 1 <= m, n <= 50
  • 0 <= maxMove <= 50
  • 0 <= startRow < m
  • 0 <= startColumn < n

Solution

Python Solution

class Solution:
    def findPaths(self, m: int, n: int, maxMove: int, startRow: int, startColumn: int) -> int:
        MOD = 10**9 + 7
        dp = {}
        
        def dfs(moves, row, col):
            if row < 0 or row >= m or col < 0 or col >= n:
                return 1
            if moves == 0:
                return 0
            
            key = (moves, row, col)
            if key in dp:
                return dp[key]
            
            paths = 0
            for dx, dy in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                paths = (paths + dfs(moves - 1, row + dx, col + dy)) % MOD
            
            dp[key] = paths
            return paths
        
        return dfs(maxMove, startRow, startColumn)

            return 1;

            return 0;

        return paths;
    }
}

JavaScript Solution

/**
 * @param {number} m
 * @param {number} n
 * @param {number} maxMove
 * @param {number} startRow
 * @param {number} startColumn
 * @return {number}
 */
var findPaths = function(m, n, maxMove, startRow, startColumn) {
    const MOD = 1e9 + 7;
    const dp = new Map();
    
    function dfs(moves, row, col) {
        if (row < 0 || row >= m || col < 0 || col >= n) {
            return 1;
        }
        if (moves === 0) {
            return 0;
        }
        
        const key = `${moves},${row},${col}`;
        if (dp.has(key)) {
            return dp.get(key);
        }
        
        let paths = 0;
        const dirs = [[0, 1], [1, 0], [0, -1], [-1, 0]];
        
        for (const [dx, dy] of dirs) {
            paths = (paths + dfs(moves - 1, row + dx, col + dy)) % MOD;
        }
        
        dp.set(key, paths);
        return paths;
    }
    
    return dfs(maxMove, startRow, startColumn);
};

C# Solution

public class Solution {
    private const int MOD = 1000000007;
    private long[,,] dp;
    
    public int FindPaths(int m, int n, int maxMove, int startRow, int startColumn) {
        dp = new long[maxMove + 1, m, n];
        for (int moves = 0; moves <= maxMove; moves++) {
            for (int i = 0; i < m; i++) {
                for (int j = 0; j < n; j++) {
                    dp[moves, i, j] = -1;
                }
            }
        }
        return (int)DFS(maxMove, startRow, startColumn, m, n);
    }
    
    private long DFS(int moves, int row, int col, int m, int n) {
        if (row < 0 || row >= m || col < 0 || col >= n) {
            return 1;
        }
        if (moves == 0) {
            return 0;
        }
        
        if (dp[moves, row, col] != -1) {
            return dp[moves, row, col];
        }
        
        long paths = 0;
        int[,] dirs = new int[,] {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
        
        for (int i = 0; i < 4; i++) {
            paths = (paths + DFS(moves - 1, row + dirs[i, 0], col + dirs[i, 1], m, n)) % MOD;
        }
        
        dp[moves, row, col] = paths;
        return paths;
    }
}
