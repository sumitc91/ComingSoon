767. Reorganize String

Jump to Solution: Python Java C++ JavaScript C#

Problem Description

Given a string s, rearrange the characters of s so that any two adjacent characters are not the same.

Return any possible rearrangement of s or return "" if not possible.

Examples:

Example 1:

Input: s = "aab"
Output: "aba"

Example 2:

Input: s = "aaab"
Output: ""

Constraints:

  • 1 ≤ s.length ≤ 500
  • s consists of lowercase English letters

Python Solution

class Solution:
    def reorganizeString(self, s: str) -> str:
        # Count frequency of each character
        freq = {}
        for c in s:
            freq[c] = freq.get(c, 0) + 1
            
        # Check if any character appears more than half the length
        max_freq = max(freq.values())
        if max_freq > (len(s) + 1) // 2:
            return ""
            
        # Create a list of tuples (frequency, character)
        chars = [(f, c) for c, f in freq.items()]
        chars.sort(reverse=True)
        
        # Create result list
        result = [None] * len(s)
        
        # Fill even positions first, then odd positions
        i = 0
        for f, c in chars:
            for _ in range(f):
                if i >= len(s):
                    i = 1
                result[i] = c
                i += 2
                
        return "".join(result)

Implementation Notes:

  • Counts frequency of each character
  • Checks if any character appears more than half the length
  • Fills even positions first, then odd positions
  • Time complexity: O(n log n) where n is length of string
  • Space complexity: O(n) for the result string

Java Solution

class Solution {
    public String reorganizeString(String s) {
        // Count frequency of each character
        int[] freq = new int[26];
        for (char c : s.toCharArray()) {
            freq[c - 'a']++;
        }
        
        // Find character with maximum frequency
        int maxFreq = 0;
        char maxChar = 'a';
        for (int i = 0; i < 26; i++) {
            if (freq[i] > maxFreq) {
                maxFreq = freq[i];
                maxChar = (char)('a' + i);
            }
        }
        
        // Check if any character appears more than half the length
        if (maxFreq > (s.length() + 1) / 2) {
            return "";
        }
        
        // Create result array
        char[] result = new char[s.length()];
        
        // Fill even positions with max frequency character
        int idx = 0;
        while (freq[maxChar - 'a'] > 0) {
            result[idx] = maxChar;
            idx += 2;
            freq[maxChar - 'a']--;
        }
        
        // Fill remaining positions with other characters
        for (int i = 0; i < 26; i++) {
            while (freq[i] > 0) {
                if (idx >= s.length()) {
                    idx = 1;
                }
                result[idx] = (char)('a' + i);
                idx += 2;
                freq[i]--;
            }
        }
        
        return new String(result);
    }
}

JavaScript Solution

function reorganizeString(s) {
    // Count frequency of each character
    const freq = new Array(26).fill(0);
    for (const c of s) {
        freq[c.charCodeAt(0) - 'a'.charCodeAt(0)]++;
    }
    
    // Find character with maximum frequency
    let maxFreq = 0;
    let maxChar = 'a';
    for (let i = 0; i < 26; i++) {
        if (freq[i] > maxFreq) {
            maxFreq = freq[i];
            maxChar = String.fromCharCode('a'.charCodeAt(0) + i);
        }
    }
    
    // Check if any character appears more than half the length
    if (maxFreq > Math.floor((s.length + 1) / 2)) {
        return "";
    }
    
    // Create result array
    const result = new Array(s.length).fill('');
    
    // Fill even positions with max frequency character
    let idx = 0;
    while (freq[maxChar.charCodeAt(0) - 'a'.charCodeAt(0)] > 0) {
        result[idx] = maxChar;
        idx += 2;
        freq[maxChar.charCodeAt(0) - 'a'.charCodeAt(0)]--;
    }
    
    // Fill remaining positions with other characters
    for (let i = 0; i < 26; i++) {
        while (freq[i] > 0) {
            if (idx >= s.length) {
                idx = 1;
            }
            result[idx] = String.fromCharCode('a'.charCodeAt(0) + i);
            idx += 2;
            freq[i]--;
        }
    }
    
    return result.join('');
}
