587. Erect the Fence

Jump to Solution: Python Java C++ JavaScript C#

Problem Description

You are given an array trees where trees[i] = [xi, yi] represents the location of a tree in the garden.

You are asked to fence the entire garden using the minimum length of rope as it is expensive. The garden is well fenced only if all the trees are enclosed. Return the coordinates of trees that are exactly located on the fence perimeter.

Example 1:

Input: trees = [[1,1],[2,2],[2,0],[2,4],[3,3],[4,2]]
Output: [[1,1],[2,0],[3,3],[2,4],[4,2]]
Explanation: All the trees will be on the perimeter of the fence.

Example 2:

Input: trees = [[1,2],[2,2],[4,2]]
Output: [[4,2],[2,2],[1,2]]
Explanation: The fence forms a line that passes through all the trees.

Constraints:

  • 1 <= points.length <= 3000
  • points[i].length == 2
  • 0 <= xi, yi <= 100
  • All the given points are unique.

Python Solution

class Solution:
    def outerTrees(self, trees: List[List[int]]) -> List[List[int]]:
        def cross(o, a, b):
            return (a[0] - o[0]) * (b[1] - o[1]) - (a[1] - o[1]) * (b[0] - o[0])
        
        if len(trees) < 4:
            return trees
            
        # Find the bottommost point
        bottom = min(trees, key=lambda p: (p[1], p[0]))
        
        # Sort points based on polar angle and distance from bottom
        trees.sort(key=lambda p: (
            math.atan2(p[1] - bottom[1], p[0] - bottom[0]),
            (p[0] - bottom[0])**2 + (p[1] - bottom[1])**2
        ))
        
        # Add bottom point to the beginning
        trees = [bottom] + trees
        
        # Build the hull
        hull = []
        for p in trees:
            while len(hull) >= 2 and cross(hull[-2], hull[-1], p) < 0:
                hull.pop()
            hull.append(p)
            
        # Remove the duplicate bottom point
        hull.pop()
        
        return hull

Java Solution

class Solution {
    private int cross(int[] o, int[] a, int[] b) {
        return (a[0] - o[0]) * (b[1] - o[1]) - (a[1] - o[1]) * (b[0] - o[0]);
    }
    
    public int[][] outerTrees(int[][] trees) {
        if (trees.length < 4) {
            return trees;
        }
        
        // Find the bottommost point
        int[] bottom = trees[0];
        for (int[] p : trees) {
            if (p[1] < bottom[1] || (p[1] == bottom[1] && p[0] < bottom[0])) {
                bottom = p;
            }
        }
        
        // Sort points based on polar angle and distance from bottom
        final int[] finalBottom = bottom;
        Arrays.sort(trees, (a, b) -> {
            double angleA = Math.atan2(a[1] - finalBottom[1], a[0] - finalBottom[0]);
            double angleB = Math.atan2(b[1] - finalBottom[1], b[0] - finalBottom[0]);
            if (angleA != angleB) {
                return Double.compare(angleA, angleB);
            }
            int distA = (a[0] - finalBottom[0]) * (a[0] - finalBottom[0]) + 
                       (a[1] - finalBottom[1]) * (a[1] - finalBottom[1]);
            int distB = (b[0] - finalBottom[0]) * (b[0] - finalBottom[0]) + 
                       (b[1] - finalBottom[1]) * (b[1] - finalBottom[1]);
            return distA - distB;
        });
        
        // Add bottom point to the beginning
        int[][] sorted = new int[trees.length + 1][];
        sorted[0] = bottom;
        System.arraycopy(trees, 0, sorted, 1, trees.length);
        
        // Build the hull
        List hull = new ArrayList<>();
        for (int[] p : sorted) {
            while (hull.size() >= 2 && cross(hull.get(hull.size() - 2), 
                                           hull.get(hull.size() - 1), p) < 0) {
                hull.remove(hull.size() - 1);
            }
            hull.add(p);
        }
        
        // Remove the duplicate bottom point
        hull.remove(hull.size() - 1);
        
        return hull.toArray(new int[0][]);
    }
}

public class Solution {
    private int Cross(int[] o, int[] a, int[] b) {
        return (a[0] - o[0]) * (b[1] - o[1]) - (a[1] - o[1]) * (b[0] - o[0]);
    }
    
    public int[][] OuterTrees(int[][] trees) {
        if (trees.Length < 4) {
            return trees;
        }
        
        // Find the bottommost point
        int[] bottom = trees[0];
        foreach (int[] p in trees) {
            if (p[1] < bottom[1] || (p[1] == bottom[1] && p[0] < bottom[0])) {
                bottom = p;
            }
        }
        
        // Sort points based on polar angle and distance from bottom
        Array.Sort(trees, (a, b) => {
            double angleA = Math.Atan2(a[1] - bottom[1], a[0] - bottom[0]);
            double angleB = Math.Atan2(b[1] - bottom[1], b[0] - bottom[0]);
            if (angleA != angleB) {
                return angleA.CompareTo(angleB);
            }
            int distA = (a[0] - bottom[0]) * (a[0] - bottom[0]) + 
                       (a[1] - bottom[1]) * (a[1] - bottom[1]);
            int distB = (b[0] - bottom[0]) * (b[0] - bottom[0]) + 
                       (b[1] - bottom[1]) * (b[1] - bottom[1]);
            return distA.CompareTo(distB);
        });
        
        // Add bottom point to the beginning
        var sorted = new int[trees.Length + 1][];
        sorted[0] = bottom;
        Array.Copy(trees, 0, sorted, 1, trees.Length);
        
        // Build the hull
        var hull = new List();
        foreach (int[] p in sorted) {
            while (hull.Count >= 2 && Cross(hull[hull.Count - 2], 
                                          hull[hull.Count - 1], p) < 0) {
                hull.RemoveAt(hull.Count - 1);
            }
            hull.Add(p);
        }
        
        // Remove the duplicate bottom point
        hull.RemoveAt(hull.Count - 1);
        
        return hull.ToArray();
    }
}
