866. Prime Palindrome

Jump to Solution: Python Java C++ JavaScript C#

Problem Description

Given an integer n, return the smallest prime palindrome greater than or equal to n.

An integer is prime if it has exactly two factors: 1 and itself.

An integer is a palindrome if it reads the same from left to right as it does from right to left.

Examples:

Example 1:

Input: n = 6
Output: 7
Explanation: 7 is the smallest prime palindrome greater than or equal to 6.

Example 2:

Input: n = 8
Output: 11
Explanation: 11 is the smallest prime palindrome greater than or equal to 8.

Example 3:

Input: n = 13
Output: 101
Explanation: 101 is the smallest prime palindrome greater than or equal to 13.

Constraints:

  • 1 ≤ n ≤ 10⁸
  • The answer is guaranteed to exist and be less than 2 * 10⁸.

Python Solution

class Solution:
    def primePalindrome(self, n: int) -> int:
        def is_prime(x):
            if x < 2:
                return False
            for i in range(2, int(x ** 0.5) + 1):
                if x % i == 0:
                    return False
            return True
        
        def next_palindrome(x):
            s = str(x)
            l = len(s)
            # Handle single digit case
            if l == 1:
                return x
            
            # Get the left half
            left_half = s[:(l+1)//2]
            num = int(left_half)
            
            while True:
                # Create palindrome from left half
                if l % 2 == 0:
                    palindrome = int(str(num) + str(num)[::-1])
                else:
                    palindrome = int(str(num) + str(num)[:-1][::-1])
                
                if palindrome >= x:
                    return palindrome
                num += 1
        
        # Handle special cases for even length palindromes
        if 8 <= n <= 11:
            return 11
        
        x = n
        while True:
            x = next_palindrome(x)
            # Skip even length palindromes > 11 as they are always divisible by 11
            if len(str(x)) % 2 == 0 and x > 11:
                x = 10 ** len(str(x))
                continue
            if is_prime(x):
                return x

Implementation Notes:

  • Uses helper functions for primality testing and palindrome generation
  • Optimizes by skipping even-length palindromes > 11 (they're always divisible by 11)
  • Time complexity: O(N^(1/4) * log N)
  • Space complexity: O(log N)

Java Solution

class Solution {
    public int primePalindrome(int n) {
        if (8 <= n && n <= 11) return 11;
        
        for (int x = n; x < 200_000_000; x++) {
            if (isPalindrome(x) && isPrime(x)) {
                // Skip even length palindromes > 11 as they are always divisible by 11
                if (String.valueOf(x).length() % 2 == 0 && x > 11) {
                    x = (int)Math.pow(10, String.valueOf(x).length());
                    x--;
                    continue;
                }
                return x;
            }
        }
        return -1;
    }
    
    private boolean isPrime(int x) {
        if (x < 2) return false;
        for (int i = 2; i <= Math.sqrt(x); i++) {
            if (x % i == 0) return false;
        }
        return true;
    }
    
    private boolean isPalindrome(int x) {
        String s = String.valueOf(x);
        int i = 0, j = s.length() - 1;
        while (i < j) {
            if (s.charAt(i++) != s.charAt(j--)) return false;
        }
        return true;
    }
}

public class Solution {
    public int PrimePalindrome(int n) {
        if (8 <= n && n <= 11) return 11;
        
        for (int x = n; x < 200_000_000; x++) {
            if (IsPalindrome(x) && IsPrime(x)) {
                // Skip even length palindromes > 11
                if (x.ToString().Length % 2 == 0 && x > 11) {
                    x = (int)Math.Pow(10, x.ToString().Length);
                    x--;
                    continue;
                }
                return x;
            }
        }
        return -1;
    }
    
    private bool IsPrime(int x) {
        if (x < 2) return false;
        for (int i = 2; i <= Math.Sqrt(x); i++) {
            if (x % i == 0) return false;
        }
        return true;
    }
    
    private bool IsPalindrome(int x) {
        string s = x.ToString();
        int i = 0, j = s.Length - 1;
        while (i < j) {
            if (s[i++] != s[j--]) return false;
        }
        return true;
    }
}
