Problem Description

You are given an integer array cards of length 4. You have four cards, each containing a number in the range [1, 9]. You should arrange the numbers on these cards in a mathematical expression using the operators ['+', '-', '*', '/'] and the parentheses '(' and ')' to get the value 24.

You are restricted to the following rules:

  • The division operator '/' represents real division, not integer division.
  • Every operation between numbers should be in floating-point arithmetic.
  • You cannot use the unary negation operator '-'.
  • You cannot concatenate numbers together, for example, if the cards are [1,2,1,2], you cannot write this as "12 + 12".

Examples:

Example 1:

Input: cards = [4,1,8,7]
Output: true
Explanation: (8-4) * (7-1) = 24

Example 2:

Input: cards = [1,2,1,2]
Output: false

Constraints:

  • cards.length == 4
  • 1 ≤ cards[i] ≤ 9

Python Solution

class Solution:
    def judgePoint24(self, cards: List[int]) -> bool:
        def solve(nums):
            if len(nums) == 1:
                return abs(nums[0] - 24) < 1e-10
            
            for i in range(len(nums)):
                for j in range(i + 1, len(nums)):
                    a, b = nums[i], nums[j]
                    remaining = nums[:i] + nums[i+1:j] + nums[j+1:]
                    
                    for op in ['+', '-', '*', '/']:
                        if op == '+':
                            remaining.append(a + b)
                        elif op == '-':
                            remaining.append(a - b)
                        elif op == '*':
                            remaining.append(a * b)
                        elif op == '/':
                            if abs(b) < 1e-10:
                                continue
                            remaining.append(a / b)
                            
                        if solve(remaining):
                            return True
                        remaining.pop()
                        
                    # Try b op a
                    for op in ['+', '-', '*', '/']:
                        if op == '+':
                            remaining.append(b + a)
                        elif op == '-':
                            remaining.append(b - a)
                        elif op == '*':
                            remaining.append(b * a)
                        elif op == '/':
                            if abs(a) < 1e-10:
                                continue
                            remaining.append(b / a)
                            
                        if solve(remaining):
                            return True
                        remaining.pop()
            
            return False
            
        return solve(cards)

Implementation Notes:

  • Uses recursive backtracking to try all possible combinations
  • Handles floating-point arithmetic with epsilon comparison
  • Time complexity: O(1) since input size is fixed
  • Space complexity: O(1) for recursion stack

Java Solution

class Solution {
    public boolean judgePoint24(int[] cards) {
        List nums = new ArrayList<>();
        for (int card : cards) {
            nums.add((double) card);
        }
        return solve(nums);
    }
    
    private boolean solve(List nums) {
        if (nums.size() == 1) {
            return Math.abs(nums.get(0) - 24) < 1e-10;
        }
        
        for (int i = 0; i < nums.size(); i++) {
            for (int j = i + 1; j < nums.size(); j++) {
                double a = nums.get(i);
                double b = nums.get(j);
                List remaining = new ArrayList<>();
                
                for (int k = 0; k < nums.size(); k++) {
                    if (k != i && k != j) {
                        remaining.add(nums.get(k));
                    }
                }
                
                // Try a op b
                for (char op : new char[]{'+', '-', '*', '/'}) {
                    if (op == '/' && Math.abs(b) < 1e-10) continue;
                    double result = 0;
                    switch (op) {
                        case '+': result = a + b; break;
                        case '-': result = a - b; break;
                        case '*': result = a * b; break;
                        case '/': result = a / b; break;
                    }
                    remaining.add(result);
                    if (solve(remaining)) return true;
                    remaining.remove(remaining.size() - 1);
                }
                
                // Try b op a
                for (char op : new char[]{'+', '-', '*', '/'}) {
                    if (op == '/' && Math.abs(a) < 1e-10) continue;
                    double result = 0;
                    switch (op) {
                        case '+': result = b + a; break;
                        case '-': result = b - a; break;
                        case '*': result = b * a; break;
                        case '/': result = b / a; break;
                    }
                    remaining.add(result);
                    if (solve(remaining)) return true;
                    remaining.remove(remaining.size() - 1);
                }
            }
        }
        return false;
    }
}
